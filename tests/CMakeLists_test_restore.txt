# Simple CMake file to build the manual restore test
cmake_minimum_required(VERSION 3.16)

# Find Qt6
find_package(Qt6 REQUIRED COMPONENTS Core Widgets)

# Add executable
add_executable(test_restore_manual test_restore_manual.cpp)

# Link Qt6 libraries
target_link_libraries(test_restore_manual 
    Qt6::Core 
    Qt6::Widgets
)

# Include directories (adjust paths as needed)
target_include_directories(test_restore_manual PRIVATE
    include
    src/core
    src/gui
)
